DROP TRIGGER DB_IUMS.CLASS_ROOM_TRIGGER;

CREATE OR REPLACE TRIGGER DB_IUMS.CLASS_ROOM_TRIGGER
BEFORE INSERT ON DB_IUMS.ROOM_INFO
FOR EACH ROW
BEGIN
  SELECT DB_IUMS.SQN_CLASS_ROOM.NEXTVAL
  INTO   :new.ROOM_ID
  FROM   dual;
END;
/
