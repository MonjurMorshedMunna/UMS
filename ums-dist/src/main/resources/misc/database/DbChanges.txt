CREATE TABLE DB_IUMS.NOTIFICATION
(
  id                 NUMBER,
  PRODUCER_ID        VARCHAR2(30),
  CONSUMER_ID        VARCHAR2(30),
  NOTIFICATION_TYPE  VARCHAR2(100),
  PAYLOAD            VARCHAR2(2048),
  PRODUCED_ON        DATE,
  CONSUMED_ON        DATE,
  LAST_MODIFIED      VARCHAR2(18)
)
RESULT_CACHE (MODE DEFAULT)
STORAGE    (
            BUFFER_POOL      DEFAULT
            FLASH_CACHE      DEFAULT
            CELL_FLASH_CACHE DEFAULT
           )
LOGGING
NOCOMPRESS
NOCACHE
NOPARALLEL
NOMONITORING
/

CREATE SEQUENCE DB_IUMS.SQN_NOTIFICATION_ID
  START WITH 1001
  MAXVALUE 9999999999999999999999999999
  MINVALUE 0
  NOCYCLE
  NOCACHE
  NOORDER;
/

CREATE OR REPLACE TRIGGER DB_IUMS.NOTIFICATION_ID_TRIGGER
BEFORE INSERT ON DB_IUMS.NOTIFICATION
FOR EACH ROW
BEGIN
  SELECT DB_IUMS.SQN_NOTIFICATION_ID.NEXTVAL
  INTO   :new.ID
  FROM   dual;
END;
